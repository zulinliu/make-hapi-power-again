```typescript
function SessionDetailRoute() {
  const { api } = useAppContext()
  const pathname = useLocation().pathname
  const { sessionId } = useParams({ from: '/sessions/$sessionId' })
  const navigate = useNavigate()
  const { session, notFound } = useSession(api, sessionId)
  const goBack = useAppGoBack()
  const [outlineOpen, setOutlineOpen] = useState(false)

  // ... existing notFound/loading handling stays ...

  const basePath = `/sessions/${sessionId}`
  const isChat = pathname === basePath || pathname === `${basePath}/`

  return (
    <div className="flex h-full min-h-0 flex-col">
      <SessionHeader
        session={session}
        onBack={goBack}
        api={api}
        sessionId={sessionId}
        isSubPage={!isChat}
        onSessionDeleted={() => navigate({ to: '/sessions' })}
        onToggleOutline={() => setOutlineOpen(prev => !prev)}
      />
      {isChat ? (
        <SessionPage outlineOpen={outlineOpen} setOutlineOpen={setOutlineOpen} />
      ) : (
        <Outlet />
      )}
    </div>
  )
}
```
